## <a name="example"></a><span data-ttu-id="0889b-109">Ejemplo</span><span class="sxs-lookup"><span data-stu-id="0889b-109">Example</span></span>

class Test  
{  
    static void Main()  
    {  
        int[] array = new int[] { 1, 2, 3, 4, 5 };
