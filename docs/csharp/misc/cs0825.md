## <a name="example"></a><span data-ttu-id="375cd-107">Ejemplo</span><span class="sxs-lookup"><span data-stu-id="375cd-107">Example</span></span>

{  
    // Both of these declarations trigger CS0825

    private var bookTitles = new List<string>();
